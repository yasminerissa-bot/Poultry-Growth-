                <p class="text-sm text-slate-600 leading-relaxed mb-3">
                    In broiler production in Malaysia, Feed Conversion Ratio (FCR) directly dictates profitability. As an animal husbandry technologist, your goal is to determine the exact peak growth efficiency day to schedule feed phase transitions (Starter $\rightarrow$ Grower $\rightarrow$ Finisher) and optimize harvest timelines.
                </p>
                <p class="text-sm text-slate-600 leading-relaxed">
                    By modeling cumulative weight as a logistic function <span class="font-semibold text-emerald-700">\(f(x)\)</span>, we can take its derivative <span class="font-semibold text-rose-600">\(f'(x)\)</span> to obtain the <strong>Average Daily Gain (ADG)</strong>. When the ADG begins to decline (past the inflection point), feed efficacy drops, signaling that continuing to feed the herd costs more than the value of weight gained.
                </p>

    <script>
        // Parameters for different housing options fitted to Malaysian environmental realities
        const models = {
            closed: {
                L: 3000,     // Cobb 500 potential under climate controlled tunnel ventilation
                k: 0.13,     // Optimal growth coefficient
                x0: 24.5,    // Day of inflection (maximum daily weight gain speed)
                name: "Closed-House (Cooling)"
            },
            open: {
                L: 2400,     // Reduced growth cap due to Malaysian heat stress & high humidity
                k: 0.11,     // Sluggish development coefficient
                x0: 27.0,    // Inflection point is delayed; chickens take longer to reach optimal weight
                name: "Open-House (Heat Stress)"
            }
        };

        let currentEnv = "closed";
        let chartInstance = null;
        let activeDay = 24;

        // Core Mathematical Formulations
        function getWeight(x, L, k, x0) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        function getDailyGain(x, L, k, x0) {
            const expTerm = Math.exp(-k * (x - x0));
            return (L * k * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // Formulating detailed advice messages depending on localized Malaysian animal husbandry factors
        function getHusbandryAdvice(day, weight, rate) {
            const L = models[currentEnv].L;
            const x0 = models[currentEnv].x0;

            if (day < 7) {
                return {
                    tag: "BROODING PHASE (STRICT TEMP CONTROL)",
                    color: "bg-blue-100 text-blue-800 border-blue-200",
                    text: `During Day ${day} (Brooding), the chicken's cumulative weight is small (${Math.round(weight)}g) and ADG is low (${rate.toFixed(1)}g/day). Focus on supplementary heating (31-33°C is required in Malaysia, despite high relative humidity) to stimulate initial gut health, organ growth, and rapid cell division.`
                };
            } else if (day >= 7 && day < 15) {
                return {
                    tag: "STARTER FEED PHASE (HIGH PROTEIN)",
                    color: "bg-indigo-100 text-indigo-800 border-indigo-200",
                    text: `Day ${day} marks the Starter Feed Phase. Weight is ~${Math.round(weight)}g. The rate of gain f'(x) is escalating rapidly to ${rate.toFixed(1)}g/day. High protein feed (21-22% Crude Protein) is highly critical to support skeletal frame construction and rapid muscle fiber multiplication.`
                };
            } else if (day >= 15 && day <= Math.floor(x0)) {
                return {
                    tag: "ACCELERATING EXPONENTIAL GROWTH",
                    color: "bg-emerald-100 text-emerald-800 border-emerald-200",
                    text: `At Day ${day}, growth is in the exponential takeoff phase, rapidly approaching the inflection point on Day ${x0}. ADG is an impressive ${rate.toFixed(1)}g/day. Metabolic heat production is critical: maintain high static pressure and strict wind speed in Closed Houses to avoid thermal exhaustion.`
                };
            } else if (day > Math.floor(x0) && day <= 32) {
                return {
                    tag: "INFLECTION DEVIATION (MAXIMUM EFFICIENCY PASSED)",
                    color: "bg-amber-100 text-amber-800 border-amber-200",
                    text: `Day ${day} has transitioned past the inflection point (Day ${x0.toFixed(1)}). The rate of daily weight gain has peaked at ~${Math.round(getDailyGain(x0, L, models[currentEnv].k, x0))}g/day and is now beginning to slow down (${rate.toFixed(1)}g/day), despite total body weight still rising. Ensure a smooth transition to Grower Feed.`
                };
            } else if (day > 32 && day < 38) {
                return {
                    tag: "FINISHER / CONSOLIDATION STAGE",
                    color: "bg-orange-100 text-orange-800 border-orange-200",
                    text: `Day ${day} represents the Finisher Phase. Cumulative weight reaches ${Math.round(weight)}g. The derivative f'(x) has dropped to ${rate.toFixed(1)}g/day. Feed formulation must transition to higher-energy/lower-protein to reduce waste, saving feed costs as bio-efficiency begins to plateau.`
                };
            } else {
                return {
                    tag: "OPTIMAL HARVEST WINDOW (FCR LIMIT REACHED)",
                    color: "bg-rose-100 text-rose-800 border-rose-200",
                    text: `At Day ${day}, the broiler weighs ${Math.round(weight)}g, but daily gain rate f'(x) has decelerated to only ${rate.toFixed(1)}g/day. Keeping birds past this day is unprofitable in Malaysia because daily feed consumption costs exceed carcass market weight value. Harvest immediately to maximize margins!`
                };
            }
        }

        // Updates all values, formulas, and text blocks on the DOM
        function updateCalculations(day) {
            activeDay = Number(day);
            const m = models[currentEnv];
            
            const w = getWeight(activeDay, m.L, m.k, m.x0);
            const r = getDailyGain(activeDay, m.L, m.k, m.x0);

            // Update DOM Values
            document.getElementById('inspector-day-badge').innerText = `Day ${activeDay}`;
            document.getElementById('calc-weight-val').innerText = w.toFixed(1);
            document.getElementById('calc-gain-val').innerText = r.toFixed(1);
            
            // Render KaTeX for specific math calculations
            const fxMathEl = document.getElementById('calc-fx-math');
            const dfxMathEl = document.getElementById('calc-dfx-math');
            
            if (window.katex) {
                katex.render(`f(${activeDay}) = \\frac{${m.L}}{1 + e^{-${m.k}(${activeDay} - ${m.x0})}} = ${w.toFixed(1)}\\text{ g}`, fxMathEl, { displayMode: false });
                katex.render(`f'(${activeDay}) = \\frac{${m.L} \\cdot ${m.k} \\cdot e^{-${m.k}(${activeDay} - ${m.x0})}}{(1 + e^{-${m.k}(${activeDay} - ${m.x0})})^2} = ${r.toFixed(1)}\\text{ g/day}`, dfxMathEl, { displayMode: false });
            }

            // Update dynamic advice
            const advice = getHusbandryAdvice(activeDay, w, r);
            document.getElementById('husbandry-advice').innerText = advice.text;
            document.getElementById('husbandry-strategic-tag').innerText = advice.tag;
            
            // Adjust alert colors
            const alertBox = document.getElementById('husbandry-alert-box');
            alertBox.className = `mt-3 flex gap-2 items-center text-xs p-2 rounded-lg border font-semibold ${advice.color}`;

            // Sync Slider position
            document.getElementById('day-slider').value = activeDay;

            // Move custom vertical highlight line in Chart
            if (chartInstance) {
                chartInstance.update('none'); // silent update without resetting animation
            }
        }

        // Main initialization of ChartJS with custom interaction support
        function initChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            const labels = Array.from({length: 43}, (_, i) => i);
            const m = models[currentEnv];

            const weightData = labels.map(day => getWeight(day, m.L, m.k, m.x0));
            const gainData = labels.map(day => getDailyGain(day, m.L, m.k, m.x0));

            // Custom vertical bar plugin to draw highlight at selected day
            const verticalLinePlugin = {
                id: 'verticalLine',
                afterDraw: (chart) => {
                    if (activeDay !== null) {
                        const ctx = chart.ctx;
                        const xVal = chart.scales.x.getPixelForValue(activeDay);
                        const topY = chart.chartArea.top;
                        const bottomY = chart.chartArea.bottom;

                        ctx.save();
                        ctx.beginPath();
                        ctx.moveTo(xVal, topY);
                        ctx.lineTo(xVal, bottomY);
                        ctx.lineWidth = 1.5;
                        ctx.setLineDash([5, 5]);
                        ctx.strokeStyle = '#475569';
                        ctx.stroke();
                        ctx.restore();
                    }
                }
            };

            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Cumulative Body Weight (g)',
                            data: weightData,
                            borderColor: '#059669', // Emerald 600
                            backgroundColor: 'rgba(5, 150, 105, 0.08)',
                            yAxisID: 'yWeight',
                            borderWidth: 3,
                            fill: true,
                            tension: 0.3,
                            pointRadius: labels.map(day => day === activeDay ? 6 : 2),
                            pointBackgroundColor: labels.map(day => day === activeDay ? '#047857' : '#10b981'),
                            pointHoverRadius: 8
                        },
                        {
                            label: 'Daily Gain ADG (g/day)',
                            data: gainData,
                            borderColor: '#e11d48', // Rose 600
                            backgroundColor: 'rgba(225, 29, 72, 0.02)',
                            yAxisID: 'yGain',
                            borderWidth: 3,
                            tension: 0.3,
                            pointRadius: labels.map(day => day === activeDay ? 6 : 2),
                            pointBackgroundColor: labels.map(day => day === activeDay ? '#be123c' : '#f43f5e'),
                            pointHoverRadius: 8
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    onClick: (e) => {
                        const points = chartInstance.getElementsAtEventForMode(e, 'index', { intersect: false }, true);
                        if (points.length > 0) {
                            const clickedIdx = points[0].index;
                            updateCalculations(clickedIdx);
                        }
                    },
                    plugins: {
                        legend: {
                            display: false // We use our custom legend in HTML
                        },
                        tooltip: {
                            mode: 'index',
                            intersect: false,
                            backgroundColor: 'rgba(15, 23, 42, 0.9)',
                            titleFont: { size: 12, weight: 'bold', family: 'Inter' },
                            bodyFont: { size: 11, family: 'Inter' },
                            padding: 10,
                            cornerRadius: 8,
                            callbacks: {
                                title: (items) => `Inspecting Day ${items[0].label}`,
                                label: (context) => {
                                    if (context.datasetIndex === 0) {
                                        return ` Weight: ${parseFloat(context.raw).toFixed(1)} g`;
                                    } else {
                                        return ` Daily Gain: ${parseFloat(context.raw).toFixed(1)} g/day`;
                                    }
                                }
                            }
                        }
                    },
                    scales: {
                        x: {
                            title: {
                                display: true,
                                text: 'Age of Broiler Herd (Days)',
                                font: { weight: 'bold', size: 11, family: 'Inter' },
                                color: '#475569'
                            },
                            grid: { color: '#f1f5f9' },
                            ticks: { color: '#64748b' }
                        },
                        yWeight: {
                            type: 'linear',
                            position: 'left',
                            title: {
                                display: true,
                                text: 'Weight f(x) (grams)',
                                font: { weight: 'bold', size: 11, family: 'Inter' },
                                color: '#059669'
                            },
                            grid: { color: '#e2e8f0' },
                            ticks: { color: '#047857' }
                        },
                        yGain: {
                            type: 'linear',
                            position: 'right',
                            title: {
                                display: true,
                                text: "Daily Gain f'(x) (g/day)",
                                font: { weight: 'bold', size: 11, family: 'Inter' },
                                color: '#e11d48'
                            },
                            grid: { drawOnChartArea: false }, // Only keep left-side grid
                            ticks: { color: '#be123c' }
                        }
                    }
                },
                plugins: [verticalLinePlugin]
            });
        }

        // Handles switching houses (Closed/Open)
        function setEnvironment(env) {
            currentEnv = env;
            
            // CSS Toggle classes
            const btnClosed = document.getElementById('btn-closed');
            const btnOpen = document.getElementById('btn-open');
            
            if (env === 'closed') {
                btnClosed.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-semibold rounded-md transition-all duration-200 bg-white text-emerald-800 shadow-sm";
                btnOpen.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-semibold rounded-md transition-all duration-200 text-slate-600 hover:text-slate-900";
            } else {
                btnOpen.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-semibold rounded-md transition-all duration-200 bg-white text-emerald-800 shadow-sm";
                btnClosed.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-semibold rounded-md transition-all duration-200 text-slate-600 hover:text-slate-900";
            }

            // Update Constant Indicators in HTML
            const m = models[currentEnv];
            document.getElementById('badge-L').innerText = `${m.L} g`;
            document.getElementById('badge-k').innerText = m.k.toFixed(3);
            document.getElementById('badge-x0').innerText = `Day ${m.x0.toFixed(1)}`;

            // Destroy and recreate chart to refresh datasets smoothly
            if (chartInstance) {
                chartInstance.destroy();
            }
            initChart();
            updateCalculations(activeDay);
        }

        // Setup initial triggers on load
        window.onload = function() {
            initChart();
            updateCalculations(24); // Start focusing around Day 24 (the critical inflection zone)
        };
